                <h2>Simplified sampling</h2>

<p style="color: #f6f6f6;">	Samples are collected accurately despite difficult conditions such as high temperatures or corrosive solutions. The small system form factor also allows the entire system to be placed in a refrigerator or climate-controlled chamber.</p> 
<p style="color: #f6f6f6;" class="box-gif-text-hide2">Samples are collected in standard HPLC or GC vials, or 96-well plates. Collection trays or plates can then be transferred directly to chromatographic system for sample analysis.</p>
<p style="color: #f6f6f6;" class="box-gif-text-hide">Flexibility of the Samplify system offers straightforward sample collection in challenging environments and applications, including high temperatures, toxic solutions, gas pipelines, and liquid process streams.</p>
            </div>
        </div>


<div style="padding:1%;" id="video">
<p class="gEnd-p">To fully understand a chemical reaction, accurate and reproducible sampling is essential. With Samplify, samples may be collected overnight or otherwise – unattended and automatically – to provide a true, representative snapshot of the reaction at any point, while offering customizable sample quench and dilution volumes. Manual sampling is thus totally eliminated. 
</p>

function smoothScroll(targetElement) {
    const targetPosition = targetElement.getBoundingClientRect().top + window.pageYOffset;
    const startPosition = window.pageYOffset;
    const distance = targetPosition - startPosition;
    const duration = 100; // Продолжительность прокрутки в миллисекундах
    let startTime = null;

    function animation(currentTime) {
        if (startTime === null) startTime = currentTime;
        const timeElapsed = currentTime - startTime;

        // Функция easeInOutQuad для ускорения и замедления
        const run = easeInOutQuad(timeElapsed, startPosition, distance, duration);
        window.scrollTo(0, run);

        if (timeElapsed < duration) requestAnimationFrame(animation);
    }

    function easeInOutQuad(t, b, c, d) {
        t /= d / 2;
        if (t < 1) return c / 2 * t * t + b;
        t=t-1;

        function smoothScrollBy(element, target, duration) {
            const start = element.scrollLeft;
            const startTime = performance.now();

            function easeInOutQuad(t, b, c, d) {
                t /= d / 2;
                if (t < 1) return c / 2 * t * t + b;
                t=t-1;
                return -c / 2 * (t * (t - 2) - 1) + b;
            }

            function animateScroll() {
                const now = performance.now();
                const elapsed = now - startTime;
                const nextPosition = easeInOutQuad(elapsed, start, target, duration);

                element.scrollLeft = nextPosition;

                if (elapsed < duration) {
                    requestAnimationFrame(animateScroll);
                } else {
                    // Устанавливаем финальную позицию, чтобы избежать возможных погрешностей
                    element.scrollLeft = start + target;
                }
            }

            animateScroll();
        }
